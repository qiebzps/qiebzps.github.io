# 用Promise对象改写三个函数

    var promiseF1 = new Promise(function(resolve,reject){
        setTimeout(() => {
            console.log("f1");
            resolve(true);
        }, 1500);
    });
    
    
    promiseF1
    .then(function(value){
        return new Promise(function(resolve,reject){
            setTimeout(() => {
                console.log("f2");
                resolve(true);
            }, 1000);
        })
    })
    .then(function(value){
            setTimeout(() => {
                console.log("f3");
            }, 500);
    })
